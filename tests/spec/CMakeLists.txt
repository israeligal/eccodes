macro(add_test_spec arg regex)
    math(EXPR TEST_COUNTER "${TEST_COUNTER} + 1")
    set(target "eccodes_tests_spec_${TEST_COUNTER}")
    ecbuild_add_test(TARGET ${target} COMMAND grib_get_gridspec ARGS ${arg} WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/../../samples)
    set_tests_properties(${target} PROPERTIES PASS_REGULAR_EXPRESSION ${regex})
endmacro()
